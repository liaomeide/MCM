import pandas as pd
data = pd.read_excel(r'D:\pythonwork\code\i_nuc.xls',index_col = 'Id',sheet_name='Sheet2')   #
outputfile = r'D:\pythonwork\code\data_type.xlsx'    #
k = 3     #
iteration = 500    #


# 

zscoredfile = r'C:\code\zscoreddata.xlsx'      #
data_zs = 1.0*(data - data.mean())/data.std()  #
data_zs.to_excel(zscoredfile, index=False)     #


#from sklearn.cluster import KMeans

#K-Means
from sklearn.cluster import KMeans
model = KMeans(n_clusters = k, n_jobs = 4, max_iter = iteration)  #k4
model.fit(data_zs)                                                 #

#
r1 = pd.Series(model.labels_).value_counts()    #
r2 = pd.DataFrame(model.cluster_centers_)       #
r = pd.concat([r2, r1], axis=1)                 #0
r.columns = list(data.columns) + [u'']   #
print(r)
#
r = pd.concat([data, pd.Series(model.labels_, index=data.index)],
              axis=1)
r.columns = list(data.columns)+[u'']     #
r.to_excel(outputfile)                           #
print(r)

def density_plot(data):                     #
    import matplotlib.pyplot as plt
    plt.rcParams['font.sans-serif']=['SimHei']      
    plt.rcParams['axes.unicode_minus'] = False     
    p = data.plot(kind='kde', linewidth = 2, subplots = True, sharex = False)
    [p[i].set_ylabel(u'') for i in range(k)]
    plt.xlabel('%s'%(i+1))
    plt.legend()
    return plt

pic_output = r'C:\code\pd_'
for i in range(k):
    density_plot(data[r[u'']==i]).savefig((u'%s%s.png') %(pic_output,i))
